255 /**
256  * Read data into the graph
257  */
258 links.Graph.prototype._readData = function(data) {
259     if (google && google.visualization && google.visualization.DataTable &&
260         data instanceof google.visualization.DataTable) {
261         // read a Google DataTable
262         this.data = [];
263 
264         for (var col = 1, cols = data.getNumberOfColumns(); col < cols; col++) {
265             var dataset = [];
266             for (var row = 0, rows = data.getNumberOfRows(); row < rows; row++) {
267                 dataset.push({"date" : data.getValue(row, 0), "value" : data.getValue(row, col)} );
268             }
269 
270             var graph = {
271                 "label": data.getColumnLabel(col),
272                 "type": undefined,
273                 "dataRange": undefined,
274                 "rowRange": undefined,
275                 "visibleRowRange": undefined,
276                 "data": dataset
277             };
278             this.data.push(graph);
279 
280             // TODO: sort by date, and remove redundant null values
281         }
282     }
283     else {
284         // parse Javascipt array
285         this.data = data || [];
286     }
287 
288     // calculate date and value ranges
289     for (var i = 0, len = this.data.length; i < len; i++) {
290         var graph = this.data[i];
291 
292         var fields = ['date'];
293         if (graph.type == 'area') {
294             fields = ['start', 'end'];
295         }
296 
297         graph.dataRange = this._getDataRange(graph.data);
298         graph.rowRange = this._getRowRange(graph.data, fields);
299     }
300 };

302 /**
303  * @constructor  links.Graph.StepDate
304  * The class StepDate is an iterator for dates. You provide a start date and an
305  * end date. The class itself determines the best scale (step size) based on the
306  * provided start Date, end Date, and minimumStep.
307  *
308  * If minimumStep is provided, the step size is chosen as close as possible
309  * to the minimumStep but larger than minimumStep. If minimumStep is not
310  * provided, the scale is set to 1 DAY.
311  * The minimumStep should correspond with the onscreen size of about 6 characters
312  *
313  * Alternatively, you can set a scale by hand.
314  * After creation, you can initialize the class by executing start(). Then you
315  * can iterate from the start date to the end date via next(). You can check if
316  * the end date is reached with the function end(). After each step, you can
317  * retrieve the current date via get().
318  * The class step has scales ranging from milliseconds, seconds, minutes, hours,
319  * days, to years.
320  *
321  * Version: 1.2
322  *
323  * @param {Date} start          The start date, for example new Date(2010, 9, 21)
324  *                              or new Date(2010, 9, 21, 23, 45, 00)
325  * @param {Date} end            The end date
326  * @param {Number}  minimumStep Optional. Minimum step size in milliseconds
327  */
328 links.Graph.StepDate = function(start, end, minimumStep) {
329 
330     // variables
331     this.current = new Date();
332     this._start = new Date();
333     this._end = new Date();
334 
335     this.autoScale  = true;
336     this.scale = links.Graph.StepDate.SCALE.DAY;
337     this.step = 1;
338 
339     // initialize the range
340     this.setRange(start, end, minimumStep);
341 };

356 /**
357  * Set a new range
358  * If minimumStep is provided, the step size is chosen as close as possible

360  * provided, the scale is set to 1 DAY.
361  * The minimumStep should correspond with the onscreen size of about 6 characters
362  * @param {Date} start        The start date and time.
363  * @param {Date} end          The end date and time.
364  * @param {int}  minimumStep  Optional. Minimum step size in milliseconds

366 links.Graph.StepDate.prototype.setRange = function(start, end, minimumStep) {
367     if (!(start instanceof Date) || !(end instanceof Date)) {

373     this._end = (end != undefined) ? new Date(end.valueOf()) : new Date();
374 
375     if (this.autoScale) {
376         this.setMinimumStep(minimumStep);
377     }
378 };

388 /**
389  * Round the current date to the first minor date value
390  * This must be executed once when the current date is set to start Date
391  */
392 links.Graph.StepDate.prototype.roundToMinor = function() {
393     // round to floor
394     // IMPORTANT: we have no breaks in this switch! (this is no bug)
395     //noinspection FallthroughInSwitchStatementJS
396     switch (this.scale) {
397         case links.Graph.StepDate.SCALE.YEAR:
398             this.current.setFullYear(this.step * Math.floor(this.current.getFullYear() / this.step));
399             this.current.setMonth(0);
400         case links.Graph.StepDate.SCALE.MONTH:        this.current.setDate(1);
401         case links.Graph.StepDate.SCALE.DAY:          // intentional fall through
402         case links.Graph.StepDate.SCALE.WEEKDAY:      this.current.setHours(0);
403         case links.Graph.StepDate.SCALE.HOUR:         this.current.setMinutes(0);
404         case links.Graph.StepDate.SCALE.MINUTE:       this.current.setSeconds(0);
405         case links.Graph.StepDate.SCALE.SECOND:       this.current.setMilliseconds(0);
406         //case links.Graph.StepDate.SCALE.MILLISECOND: // nothing to do for milliseconds
407     }
408 
409     if (this.step != 1) {
410         // round down to the first minor value that is a multiple of the current step size
411         switch (this.scale) {
412             case links.Graph.StepDate.SCALE.MILLISECOND:  this.current.setMilliseconds(this.current.getMilliseconds() - this.current.getMilliseconds() % this.step);  break;
413             case links.Graph.StepDate.SCALE.SECOND:       this.current.setSeconds(this.current.getSeconds() - this.current.getSeconds() % this.step); break;
414             case links.Graph.StepDate.SCALE.MINUTE:       this.current.setMinutes(this.current.getMinutes() - this.current.getMinutes() % this.step); break;
415             case links.Graph.StepDate.SCALE.HOUR:         this.current.setHours(this.current.getHours() - this.current.getHours() % this.step); break;
416             case links.Graph.StepDate.SCALE.WEEKDAY:      // intentional fall through
417             case links.Graph.StepDate.SCALE.DAY:          this.current.setDate((this.current.getDate()-1) - (this.current.getDate()-1) % this.step + 1); break;
418             case links.Graph.StepDate.SCALE.MONTH:        this.current.setMonth(this.current.getMonth() - this.current.getMonth() % this.step);  break;
419             case links.Graph.StepDate.SCALE.YEAR:         this.current.setFullYear(this.current.getFullYear() - this.current.getFullYear() % this.step); break;
420             default: break;
421         }
422     }
423 };

433 /**
434  * Do the next step
435  */
436 links.Graph.StepDate.prototype.next = function() {
437     var prev = this.current.valueOf();
438 
439     // Two cases, needed to prevent issues with switching daylight savings
440     // (end of March and end of October)
441     if (this.current.getMonth() < 6)   {
442         switch (this.scale) {
443             case links.Graph.StepDate.SCALE.MILLISECOND:
444 
445                 this.current = new Date(this.current.valueOf() + this.step); break;
446             case links.Graph.StepDate.SCALE.SECOND:       this.current = new Date(this.current.valueOf() + this.step * 1000); break;
447             case links.Graph.StepDate.SCALE.MINUTE:       this.current = new Date(this.current.valueOf() + this.step * 1000 * 60); break;
448             case links.Graph.StepDate.SCALE.HOUR:
449                 this.current = new Date(this.current.valueOf() + this.step * 1000 * 60 * 60);
450                 // in case of skipping an hour for daylight savings, adjust the hour again (else you get: 0h 5h 9h ... instead of 0h 4h 8h ...)
451                 var h = this.current.getHours();
452                 this.current.setHours(h - (h % this.step));
453                 break;
454             case links.Graph.StepDate.SCALE.WEEKDAY:      // intentional fall through
455             case links.Graph.StepDate.SCALE.DAY:          this.current.setDate(this.current.getDate() + this.step); break;
456             case links.Graph.StepDate.SCALE.MONTH:        this.current.setMonth(this.current.getMonth() + this.step); break;
457             case links.Graph.StepDate.SCALE.YEAR:         this.current.setFullYear(this.current.getFullYear() + this.step); break;
458             default:                      break;
459         }
460     }
461     else {
462         switch (this.scale) {
463             case links.Graph.StepDate.SCALE.MILLISECOND:  this.current = new Date(this.current.valueOf() + this.step); break;
464             case links.Graph.StepDate.SCALE.SECOND:       this.current.setSeconds(this.current.getSeconds() + this.step); break;
465             case links.Graph.StepDate.SCALE.MINUTE:       this.current.setMinutes(this.current.getMinutes() + this.step); break;
466             case links.Graph.StepDate.SCALE.HOUR:         this.current.setHours(this.current.getHours() + this.step); break;
467             case links.Graph.StepDate.SCALE.WEEKDAY:      // intentional fall through
468             case links.Graph.StepDate.SCALE.DAY:          this.current.setDate(this.current.getDate() + this.step); break;
469             case links.Graph.StepDate.SCALE.MONTH:        this.current.setMonth(this.current.getMonth() + this.step); break;
470             case links.Graph.StepDate.SCALE.YEAR:         this.current.setFullYear(this.current.getFullYear() + this.step); break;
471             default:                      break;
472         }
473     }
474 
475     if (this.step != 1) {
476         // round down to the correct major value
477         switch (this.scale) {
478             case links.Graph.StepDate.SCALE.MILLISECOND:  if(this.current.getMilliseconds() < this.step) this.current.setMilliseconds(0);  break;
479             case links.Graph.StepDate.SCALE.SECOND:       if(this.current.getSeconds() < this.step) this.current.setSeconds(0);  break;
480             case links.Graph.StepDate.SCALE.MINUTE:       if(this.current.getMinutes() < this.step) this.current.setMinutes(0);  break;
481             case links.Graph.StepDate.SCALE.HOUR:         if(this.current.getHours() < this.step) this.current.setHours(0);  break;
482             case links.Graph.StepDate.SCALE.WEEKDAY:      // intentional fall through
483             case links.Graph.StepDate.SCALE.DAY:          if(this.current.getDate() < this.step+1) this.current.setDate(1); break;
484             case links.Graph.StepDate.SCALE.MONTH:        if(this.current.getMonth() < this.step) this.current.setMonth(0);  break;
485             case links.Graph.StepDate.SCALE.YEAR:         break; // nothing to do for year
486             default:                break;
487         }
488     }
489 
490     // safety mechanism: if current time is still unchanged, move to the end
491     if (this.current.valueOf() == prev) {
492         this.current = new Date(this._end.valueOf());
493     }
494 };

537 /**
538  * Automatically determine the scale that bests fits the provided minimum step
539  * @param {Number} minimumStep  The minimum step size in milliseconds
540  */
541 links.Graph.StepDate.prototype.setMinimumStep = function(minimumStep) {
542     if (minimumStep == undefined) {
543         return;
544     }
545 
546     var stepYear       = (1000 * 60 * 60 * 24 * 30 * 12);
547     var stepMonth      = (1000 * 60 * 60 * 24 * 30);
548     var stepDay        = (1000 * 60 * 60 * 24);
549     var stepHour       = (1000 * 60 * 60);
550     var stepMinute     = (1000 * 60);
551     var stepSecond     = (1000);
552     var stepMillisecond= (1);
553 
554     // find the smallest step that is larger than the provided minimumStep
555     if (stepYear*1000 > minimumStep)        {this.scale = links.Graph.StepDate.SCALE.YEAR;        this.step = 1000;}
556     if (stepYear*500 > minimumStep)         {this.scale = links.Graph.StepDate.SCALE.YEAR;        this.step = 500;}
557     if (stepYear*100 > minimumStep)         {this.scale = links.Graph.StepDate.SCALE.YEAR;        this.step = 100;}
558     if (stepYear*50 > minimumStep)          {this.scale = links.Graph.StepDate.SCALE.YEAR;        this.step = 50;}
559     if (stepYear*10 > minimumStep)          {this.scale = links.Graph.StepDate.SCALE.YEAR;        this.step = 10;}
560     if (stepYear*5 > minimumStep)           {this.scale = links.Graph.StepDate.SCALE.YEAR;        this.step = 5;}
561     if (stepYear > minimumStep)             {this.scale = links.Graph.StepDate.SCALE.YEAR;        this.step = 1;}
562     if (stepMonth*3 > minimumStep)          {this.scale = links.Graph.StepDate.SCALE.MONTH;       this.step = 3;}

586 /**
587  * Snap a date to a rounded value. The snap intervals are dependent on the
588  * current scale and step.
589  * @param {Date} date   the date to be snapped
590  */
591 links.Graph.StepDate.prototype.snap = function(date) {
592     if (this.scale == links.Graph.StepDate.SCALE.YEAR) {
593         var year = date.getFullYear() + Math.round(date.getMonth() / 12);
594         date.setFullYear(Math.round(year / this.step) * this.step);
595         date.setMonth(0);
596         date.setDate(0);
597         date.setHours(0);
598         date.setMinutes(0);
599         date.setSeconds(0);
600         date.setMilliseconds(0);
601     }
602     else if (this.scale == links.Graph.StepDate.SCALE.MONTH) {
603         if (date.getDate() > 15) {
604             date.setDate(1);
605             date.setMonth(date.getMonth() + 1);
606             // important: first set Date to 1, after that change the month.
607         }
608         else {
609             date.setDate(1);
610         }
611 
612         date.setHours(0);
613         date.setMinutes(0);
614         date.setSeconds(0);
615         date.setMilliseconds(0);
616     }
617     else if (this.scale == links.Graph.StepDate.SCALE.DAY ||
618         this.scale == links.Graph.StepDate.SCALE.WEEKDAY) {
619         switch (this.step) {
620             case 5:
621             case 2:
622                 date.setHours(Math.round(date.getHours() / 24) * 24); break;
623             default:
624                 date.setHours(Math.round(date.getHours() / 12) * 12); break;
625         }
626         date.setMinutes(0);
627         date.setSeconds(0);
628         date.setMilliseconds(0);
629     }
630     else if (this.scale == links.Graph.StepDate.SCALE.HOUR) {
631         switch (this.step) {
632             case 4:
633                 date.setMinutes(Math.round(date.getMinutes() / 60) * 60); break;
634             default:
635                 date.setMinutes(Math.round(date.getMinutes() / 30) * 30); break;
636         }
637         date.setSeconds(0);
638         date.setMilliseconds(0);
639     } else if (this.scale == links.Graph.StepDate.SCALE.MINUTE) {
640         switch (this.step) {
641             case 15:
642             case 10:
643                 date.setMinutes(Math.round(date.getMinutes() / 5) * 5);
644                 date.setSeconds(0);
645                 break;
646             case 5:
647                 date.setSeconds(Math.round(date.getSeconds() / 60) * 60); break;
648             default:
649                 date.setSeconds(Math.round(date.getSeconds() / 30) * 30); break;
650         }
651         date.setMilliseconds(0);
652     }
653     else if (this.scale == links.Graph.StepDate.SCALE.SECOND) {
654         switch (this.step) {
655             case 15:
656             case 10:
657                 date.setSeconds(Math.round(date.getSeconds() / 5) * 5);
658                 date.setMilliseconds(0);
659                 break;
660             case 5:
661                 date.setMilliseconds(Math.round(date.getMilliseconds() / 1000) * 1000); break;
662             default:
663                 date.setMilliseconds(Math.round(date.getMilliseconds() / 500) * 500); break;
664         }
665     }
666     else if (this.scale == links.Graph.StepDate.SCALE.MILLISECOND) {
667         var step = this.step > 5 ? this.step / 2 : 1;
668         date.setMilliseconds(Math.round(date.getMilliseconds() / step) * step);
669     }
670 };

801 /**
802  * @class StepNumber
803  * The class StepNumber is an iterator for numbers. You provide a start and end
804  * value, and a best step size. StepNumber itself rounds to fixed values and
805  * a finds the step that best fits the provided step.
806  *
807  * If prettyStep is true, the step size is chosen as close as possible to the
808  * provided step, but being a round value like 1, 2, 5, 10, 20, 50, ....
809  *
810  * Example usage:
811  *   var step = new links.Graph.StepNumber(0, 10, 2.5, true);
812  *   step.start();
813  *   while (!step.end()) {

819  *

821  * @param {number} end         The end value
822  * @param {number} step        Optional. Step size. Must be a positive value.
823  * @param {boolean} prettyStep Optional. If true, the step size is rounded
824  *                             To a pretty step size (like 1, 2, 5, 10, 20, 50, ...)
825  */

853 /**
854  * Set a new step size
855  * @param {number} step        New step size. Must be a positive value
856  * @param {boolean} prettyStep Optional. If true, the provided step is rounded
857  *                             to a pretty step size (like 1, 2, 5, 10, 20, 50, ...)
858  */
859 links.Graph.StepNumber.prototype.setStep = function(step, prettyStep) {
860     if (step == undefined || step <= 0)
861         return;
862 
863     this.prettyStep = prettyStep;
864     if (this.prettyStep == true)
865         this._step = links.Graph.StepNumber._calculatePrettyStep(step);
866     else
867         this._step = step;
868 
869 
870     if (this._end / this._step > Math.pow(10, this.precision)) {
871         this.precision = undefined;
872     }

875 /**
876  * Calculate a nice step size, closest to the desired step size.
877  * Returns a value in one of the ranges 1*10^n, 2*10^n, or 5*10^n, where n is an
878  * integer number. For example 1, 2, 5, 10, 20, 50, etc...
879  * @param {number}  step  Desired step size
880  * @return {number}       Nice step size
881  */
882 links.Graph.StepNumber._calculatePrettyStep = function (step) {
883     log10 = function (x) {return Math.log(x) / Math.LN10;};
884 
885     // try three steps (multiple of 1, 2, or 5
886     var step1 = 1 * Math.pow(10, Math.round(log10(step / 1)));
887     var step2 = 2 * Math.pow(10, Math.round(log10(step / 2)));
888     var step5 = 5 * Math.pow(10, Math.round(log10(step / 5)));
889 
890     // choose the best step (closest to minimum step)
891     var prettyStep = step1;
892     if (Math.abs(step2 - step) <= Math.abs(prettyStep - step)) prettyStep = step2;
893     if (Math.abs(step5 - step) <= Math.abs(prettyStep - step)) prettyStep = step5;
894 
895     // for safety
896     if (prettyStep <= 0) {
897         prettyStep = 1;
898     }
899 
900     return prettyStep;
901 };

1237 /**
1238  * Apply a visible range. The range is limited to feasible maximum and minimum
1239  * range.

1242  * @param {Date}   zoomAroundDate  Optional. Date around which will be zoomed.
1243  */
1244 links.Graph.prototype._applyRange = function (start, end, zoomAroundDate) {
1245     // calculate new start and end value
1246     var startValue = start.valueOf();
1247     var endValue = end.valueOf();
1248     var interval = (endValue - startValue);
1249 
1250     // determine maximum and minimum interval
1251     var year = 1000 * 60 * 60 * 24 * 365;
1252     var zoomMin = Number(this.zoomMin) || 10;
1253     if (zoomMin < 10) {
1254         zoomMin = 10;
1255     }
1256     var zoomMax = Number(this.zoomMax) || 10000 * year;
1257     if (zoomMax > 10000 * year) {
1258         zoomMax = 10000 * year;
1259     }
1260     if (zoomMax < zoomMin) {
1261         zoomMax = zoomMin;
1262     }
1263 
1264     // determine min and max date value
1265     var min = this.min ? this.min.valueOf() : undefined;
1266     var max = this.max ? this.max.valueOf() : undefined;
1267     if (min && max) {
1268         if (min >= max) {
1269             // empty range
1270             var day = 1000 * 60 * 60 * 24;
1271             max = min + day;
1272         }
1273         if (zoomMax > (max - min)) {
1274             zoomMax = (max - min);
1275         }
1276         if (zoomMin > (max - min)) {
1277             zoomMin = (max - min);
1278         }
1279     }
1280 
1281     // prevent empty interval
1282     if (startValue >= endValue) {
1283         endValue += 1000 * 60 * 60 * 24;
1284     }
1285 
1286     // prevent too small scale
1287     // TODO: IE has problems with milliseconds
1288     if (interval < zoomMin) {
1289         var diff = (zoomMin - interval);
1290         var f = zoomAroundDate ? (zoomAroundDate.valueOf() - startValue) / interval : 0.5;
1291         startValue -= Math.round(diff * f);
1292         endValue   += Math.round(diff * (1 - f));
1293     }
1294 
1295     // prevent too large scale
1296     if (interval > zoomMax) {
1297         var diff = (interval - zoomMax);
1298         var f = zoomAroundDate ? (zoomAroundDate.valueOf() - startValue) / interval : 0.5;
1299         startValue += Math.round(diff * f);
1300         endValue   -= Math.round(diff * (1 - f));
1301     }
1302 
1303     // prevent to small start date
1304     if (min) {
1305         var diff = (startValue - min);
1306         if (diff < 0) {
1307             startValue -= diff;
1308             endValue -= diff;
1309         }
1310     }
1311 
1312     // prevent to large end date
1313     if (max) {
1314         var diff = (max - endValue);
1315         if (diff < 0) {
1316             startValue += diff;
1317             endValue += diff;
1318         }
1319     }

1322     this.start = new Date(startValue);
